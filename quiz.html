    <script>

        // Define the correct answers for the quiz

        const correctAnswers = {

            q1: 'multiplexing', // Fill-in-the-blank (case insensitive)

            q2: 'C', // Multiple Choice

            q3: 'A', // Multiple Choice

            q4: 'A', // Multiple Choice

            q5: ['A', 'D'] // Multi-select (must match all)

        };


        const quizForm = document.getElementById('http-quiz-form');

        const resultsSummary = document.getElementById('results-summary');

        const overallResult = document.getElementById('overall-result');

        const finalScoreDisplay = document.getElementById('final-score');

        const feedbackMessage = document.getElementById('feedback-message');

        const resetBtn = document.getElementById('reset-btn');


        // Total number of questions for score calculation

        const totalQuestions = 5;

        // Passing threshold (80% for 5 questions = 4 points)

        const passingScore = 4;


        /**

         * Checks the user's answers, calculates the score, and displays results.

         * @param {Event} e - The form submission event.

         */

        quizForm.addEventListener('submit', function(e) {

            e.preventDefault(); // Prevent default form submission

            let score = 0;


            // Function to check Fill-in-the-Blank (Q1)

            const q1Input = document.getElementById('q1-blank');

            const q1Result = document.getElementById('result-q1');

            const q1Answer = q1Input.value.trim().toLowerCase();

            const q1Correct = correctAnswers.q1.toLowerCase();


            if (q1Answer === q1Correct) {

                score++;

                displayQuestionResult(q1Result, true, q1Correct);

            } else {

                displayQuestionResult(q1Result, false, q1Correct);

            }

            q1Input.disabled = true;


            // Function to check Radio Button Questions (Q2, Q3, Q4)

            ['q2', 'q3', 'q4'].forEach(qName => {

                const selected = document.querySelector(`input[name="${qName}"]:checked`);

                const qResult = document.getElementById(`result-${qName}`);

                const isCorrect = selected && selected.value === correctAnswers[qName];


                if (isCorrect) {

                    score++;

                }

                displayQuestionResult(qResult, isCorrect, correctAnswers[qName]);


                // Disable all radio buttons for the submitted question

                document.querySelectorAll(`input[name="${qName}"]`).forEach(input => input.disabled = true);

            });


            // Function to check Multi-Select Question (Q5)

            const q5Checkboxes = document.querySelectorAll('input[name="q5"]:checked');

            const q5SelectedValues = Array.from(q5Checkboxes).map(cb => cb.value);

            const q5Correct = correctAnswers.q5;

            const q5Result = document.getElementById('result-q5');


            // Check if selected answers exactly match the correct answers

            const isQ5Correct = q5SelectedValues.length === q5Correct.length &&

                                q5Correct.every(val => q5SelectedValues.includes(val));


            if (isQ5Correct) {

                score++;

            }

            displayQuestionResult(q5Result, isQ5Correct, q5Correct.join(' & '));


            // Disable all checkboxes for Q5

            document.querySelectorAll('input[name="q5"]').forEach(input => input.disabled = true);


            // Display Summary Results

            finalScoreDisplay.textContent = score;

            resultsSummary.classList.remove('hidden');


            const isPassing = score >= passingScore;

            overallResult.textContent = isPassing ? '🎉 Congratulations! You Passed!' : '😢 You Did Not Pass.';

            overallResult.classList.remove('pass-text', 'fail-text', 'bg-red-500', 'bg-green-500');

            overallResult.classList.add(isPassing ? 'pass-text' : 'fail-text', isPassing ? 'bg-green-500' : 'bg-red-500');


            feedbackMessage.textContent = isPassing ?

                "You have a solid understanding of key HTTP concepts." :

                "Review the materials and try again to master the fundamentals.";


            document.getElementById('submit-btn').disabled = true;

        });



        /**

         * Helper function to display the result (correct/incorrect) and the correct answer for a single question.

         * @param {HTMLElement} element - The result display element for the question.

         * @param {boolean} isCorrect - Whether the user answered correctly.

         * @param {string|string[]} correctAnswer - The correct answer(s) to display.

         */

        function displayQuestionResult(element, isCorrect, correctAnswer) {

            element.classList.remove('correct-answer', 'incorrect-answer');

            element.classList.add(isCorrect ? 'correct-answer' : 'incorrect-answer');


            let resultText = isCorrect ? '✅ Correct.' : '❌ Incorrect.';

            

            // Only show the correct answer if the user was incorrect

            if (!isCorrect) {

                resultText += ` The correct answer is: ${Array.isArray(correctAnswer) ? correctAnswer.join(' and ') : correctAnswer}.`;

            }


            element.textContent = resultText;

        }



        /**

         * Resets the entire quiz state: inputs, scores, and results.

         */

        resetBtn.addEventListener('click', function() {

            quizForm.reset(); // Clears all form inputs

            resultsSummary.classList.add('hidden'); // Hide summary


            // Re-enable submit button

            document.getElementById('submit-btn').disabled = false;


            // Clear and reset question results and inputs

            document.querySelectorAll('[id^="result-q"]').forEach(el => el.textContent = '');

            document.querySelectorAll('input').forEach(input => input.disabled = false);

        });


    </script>
